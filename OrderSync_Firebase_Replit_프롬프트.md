interface ParsedItem {
  original: string;
  product: string;
  position: 'ì „ë°©' | 'í›„ë°©';
  side: 'LH' | 'RH';
  quantity: number;
}

export function parseOrderLine(line: string): ParsedItem[] {
  const result: ParsedItem[] = [];
  const tokens = line.trim().split(/\s+/);
  
  if (tokens.length === 0) return result;
  
  // ì œí’ˆëª…
  let product = tokens[0];
  const remaining = tokens.slice(1).join(' ');
  
  // ìœ„ì¹˜ ê°ì§€
  const isRear = product.includes('í›„') || remaining.includes('í›„');
  const position = isRear ? 'í›„ë°©' : 'ì „ë°©';
  product = product.replace('í›„', '');
  
  // ì¸¡ë©´ ê°ì§€
  let side: 'LH' | 'RH' | null = null;
  if (remaining.match(/\bR\b/i)) side = 'RH';
  else if (remaining.match(/\bL\b/i)) side = 'LH';
  
  // ìˆ˜ëŸ‰ íŒŒì‹±
  // íŒ¨í„´ 1: 6/42
  const slashMatch = remaining.match(/(\d+)\s*\/\s*(\d+)/);
  if (slashMatch) {
    result.push({
      original: line,
      product,
      position,
      side: 'LH',
      quantity: parseInt(slashMatch[1])
    });
    result.push({
      original: line,
      product,
      position,
      side: 'RH',
      quantity: parseInt(slashMatch[2])
    });
    return result;
  }
  
  // íŒ¨í„´ 2: 20ëŒ€ë¶„
  const setMatch = remaining.match(/(\d+)\s*ëŒ€ë¶„/);
  if (setMatch) {
    const qty = parseInt(setMatch[1]);
    result.push({
      original: line,
      product,
      position,
      side: 'LH',
      quantity: qty
    });
    result.push({
      original: line,
      product,
      position,
      side: 'RH',
      quantity: qty
    });
    return result;
  }
  
  // íŒ¨í„´ 3: R 15, L 30, ë˜ëŠ” ìˆ«ìë§Œ
  const qtyMatch = remaining.match(/(\d+)/);
  if (qtyMatch) {
    const qty = parseInt(qtyMatch[1]);
    const finalSide = side || 'RH';
    result.push({
      original: line,
      product,
      position,
      side: finalSide,
      quantity: qty
    });
  }
  
  return result;
}

export function parseOrderText(text: string): ParsedItem[] {
  const lines = text.split('\n').filter(line => line.trim());
  const results: ParsedItem[] = [];
  
  lines.forEach(line => {
    const parsed = parseOrderLine(line);
    results.push(...parsed);
  });
  
  return results;
}
